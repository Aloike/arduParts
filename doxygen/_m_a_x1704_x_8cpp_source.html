160 float MAX1704X::cellVoltage_V()
161 {
163  float resolution_mV = 0;
164 
165  /* Select the resolution depending on the part type */
166  switch( this->m_partType )
167  {
168  case PartMAX17043:
169  resolution_mV = 1.25; /*< the MAX17043's ADC has a 1.25mV resolution */
170  break;
171 
172  case PartMAX17044:
173  resolution_mV = 2.50; /*< the MAX17044's ADC has a 2.50mV resolution */
174  break;
175 
176  default:
177  break;
178  }
179 
180  /* Calculate and return the cell voltage */
181  return ( ( this->read_u16( MAX1704X_VCELL_H_REGADDR )
182  >> 4 ) /*< The last 4 bits of the 16bits registers are unused */
183  * resolution_mV )
184  / 1000.0; /*< conversion mV to Volts */
185 }

316 bool MAX1704X::setAlertThreshold_pc(uint8_t argPercents)
317 {
318  if( argPercents > 32U ) {
319  return false;
320  }
321 
322 
323  /* Calculate two's complement for threshold new value */
324  uint16_t threshold_tc = 32U - argPercents;
325 // Serial.print( String( "threshold_tc==" ) + threshold_tc );

327  /* Read CONFIG register's content */

334 // Serial.print( String( "reg_content==" ) );
335 // Serial.println( reg_content, BIN );

342  /* write register's new value */
