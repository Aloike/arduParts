729 bool Mpu6050::init(String *argErrorString)
730 {
731  uint8_t c;
732  int error = 0;
733 
734 
735  /* Initialize the 'Wire' class for the I2C-bus. */
736  LOGLN( "Init Wire..............................." );
737  Wire.begin();
738  LOGLN( "Init Wire.......................[ OK ]" );
739 
740 
741  // default at power-up:
742  // Gyro at 250 degrees second
743  // Acceleration at 2g
744  // Clock source at internal 8MHz
745  // The device is in sleep mode.
746  //
747 
748  LOGLN( "Reading WhoAmI.........................." );
749  error = this->read( MPU6050_WHO_AM_I, &this->m_whoAmI, 1 );
750  if( error != 0 ) {
751  LOGLN( "Reading WhoAmI..................[ERREUR]" );
752  if( argErrorString != 0 ) {
753  *argErrorString = String( "Can't fetch who i am" );
754  }
755  return false;
756  }
757  LOGLN( "Reading WhoAmI..................[ OK ]" );
758 
759  // According to the datasheet, the 'sleep' bit
760  // should read a '1'. But I read a '0'.
761  // That bit has to be cleared, since the sensor
762  // is in sleep mode at power-up. Even if the
763  // bit reads '0'.
764  LOGLN( "Reading PowerManagement................." );
765  error = this->read( MPU6050_PWR_MGMT_2, &c, 1 );
766  if( error != 0 ) {
767  if( argErrorString != 0 ) {
768  *argErrorString = String( "Can't fetch sleep bit" );
769  }
770  LOGLN( "Reading PowerManagement.........[ERREUR]" );
771  return false;
772  }
773  LOGLN( "Reading PowerManagement.........[ OK ]" );
774 
775 
776  /* Clear the 'sleep' bit to start the sensor. */
777  LOGLN( "Clear 'sleep' bit......................." );
778  this->write_reg( MPU6050_PWR_MGMT_1, 0 );
779  LOGLN( "Clear 'sleep' bit...............[ OK ]" );
780 
781 
782  /* Get the accelerometer's config */
783  LOGLN( "Get the accelerometer config............" );
784  uint8_t reg_content;
785  this->read( MPU6050_ACCEL_CONFIG, &reg_content, 1 );
786 
787 
788  if( ( reg_content & MPU6050_AFS_SEL_16G ) == MPU6050_AFS_SEL_16G ) {
789  this->m_accelerometerRange = AccelerometerRange_16g;
790  } else if( ( reg_content & MPU6050_AFS_SEL_8G ) == MPU6050_AFS_SEL_8G ) {
791  this->m_accelerometerRange = AccelerometerRange_8g;
792  } else if( ( reg_content & MPU6050_AFS_SEL_4G ) == MPU6050_AFS_SEL_4G ) {
793  this->m_accelerometerRange = AccelerometerRange_4g;
794  } else if( (reg_content & MPU6050_AFS_SEL_2G) == MPU6050_AFS_SEL_2G ) {
795  this->m_accelerometerRange = AccelerometerRange_2g;
796  }
797  LOGLN( "Get the accelerometer config....[ OK ]" );
798 
799 
800  /* Get the gyro config */
801  LOGLN( "Get the gyro config....................." );
802  this->read( MPU6050_GYRO_CONFIG, &reg_content, 1 );
803  if( (reg_content & MPU6050_FS_SEL_2000) == MPU6050_FS_SEL_2000 ) {
804  this->m_gyroRange = GyroRange2000deg_s;
805  } else if( (reg_content & MPU6050_FS_SEL_1000) == MPU6050_FS_SEL_1000 ) {
806  this->m_gyroRange = GyroRange1000deg_s;
807  } else if( (reg_content & MPU6050_FS_SEL_500) == MPU6050_FS_SEL_500 ) {
808  this->m_gyroRange = GyroRange500deg_s;
809  } else if( (reg_content & MPU6050_FS_SEL_250) == MPU6050_FS_SEL_250 ) {
810  this->m_gyroRange = GyroRange250deg_s;
811  }
812  LOGLN( "Get the gyro config.............[ OK ]" );
813 
814 
815 
816 // Serial.print( "MPU6050_ACCEL_CONFIG register == 0b" );
817 // Serial.print( reg_content, BIN );
818 // Serial.println( "" );
819 
820  return true;
821 }

899 bool Mpu6050::updateData()
900 {
901  int error = 0;
902 
903  // Read the raw values.
904  // Read 14 bytes at once,
905  // containing acceleration, temperature and gyro.
906  // With the default settings of the MPU-6050,
907  // there is no filter enabled, and the values
908  // are not very stable.
909  error = this->read( MPU6050_ACCEL_XOUT_H,
910  (uint8_t *) &this->m_data,
911  sizeof( this->m_data ) );
912  if( error != 0 ) {
913  Serial.print(F(" Mpu6050::updateData() ==>reading accel, temp and gyro, error = "));
914  Serial.println(error,DEC);
915 
916  this->clearData();
917  return false;
918  }
919 
920 
921  // Swap all high and low bytes.
922  // After this, the registers values are swapped,
923  // so the structure name like x_accel_l does no
924  // longer contain the lower byte.
925  uint8_t swap;
926 #define SWAP(x,y) swap = x; x = y; y = swap
927 
928  SWAP (this->m_data.reg.x_accel_h, this->m_data.reg.x_accel_l);
929  SWAP (this->m_data.reg.y_accel_h, this->m_data.reg.y_accel_l);
930  SWAP (this->m_data.reg.z_accel_h, this->m_data.reg.z_accel_l);
931  SWAP (this->m_data.reg.t_h, this->m_data.reg.t_l);
932  SWAP (this->m_data.reg.x_gyro_h, this->m_data.reg.x_gyro_l);
933  SWAP (this->m_data.reg.y_gyro_h, this->m_data.reg.y_gyro_l);
934  SWAP (this->m_data.reg.z_gyro_h, this->m_data.reg.z_gyro_l);
935 
936 
937 
938  /* calculating force vector's length (for angular calculus) */
939  this->m_forceVector = sqrt( pow( this->accelX_g(), 2 )
940  + pow( this->accelY_g(), 2 )
941  + pow( this->accelZ_g(), 2 ) );
942 
943  return true;
944 }
